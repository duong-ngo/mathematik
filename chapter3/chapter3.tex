\chapter{Định thức và hệ phương trình tuyến tính}

\par Thực hiện các phép nhân sau đây, viết các phép thế thu được thành tích của những xích rời rạc và tính dấu của chúng.

\begin{exercise}
    $
        \begin{pmatrix}
            1 & 2 & 3 & 4 & 5 \\
            2 & 4 & 5 & 1 & 3
        \end{pmatrix}
        \begin{pmatrix}
            1 & 2 & 3 & 4 & 5 \\
            4 & 3 & 5 & 1 & 2
        \end{pmatrix}
    $.
\end{exercise}

\begin{exercise}
    $
        \begin{pmatrix}
            1 & 2 & 3 & 4 & 5 \\
            3 & 5 & 4 & 1 & 2
        \end{pmatrix}
        \begin{pmatrix}
            1 & 2 & 3 & 4 & 5 \\
            4 & 3 & 1 & 5 & 2
        \end{pmatrix}
    $.
\end{exercise}

\begin{exercise}
    $(1,2)(2,3)\cdots (n-1,n)$.
\end{exercise}

\begin{exercise}
    $(1,2,3)(2,3,4)(3,4,5)\cdots (n-2,n-1,n)$.
\end{exercise}
